{% trans "Quick start (curl)" %}

{% trans "Authenticate with your key, create a delivery, then start it. Replace javi_live_… with your key." %} 

# 1) {% trans "Create a delivery" %}
curl -X POST https://javi.serbito.rs/api/v1/deliveries \
  -H "Authorization: Bearer javi_live_…" \
  -H "Content-Type: application/json" \
  -H "Idempotency-Key: order-1042" \
  -d '{"recipient_name":"Ana","recipient_phone":"064 123 4567","address":"Knez Mihailova 6, Beograd"}'
# → 201 {"id":1,"status":"pending","status_internal":"new", ...}

# 2) {% trans "Start the delivery (notifies the customer)" %}
curl -X POST https://javi.serbito.rs/api/v1/deliveries/1/start \
  -H "Authorization: Bearer javi_live_…"
# → 200 {"status":"out_for_delivery","eta":"16:30","tracking_url":"https://javi.serbito.rs/t/…"}
